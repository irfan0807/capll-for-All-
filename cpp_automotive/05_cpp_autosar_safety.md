## 3. AUTOSAR E2E Protection (End-to-End)

```cpp
// E2E Profile 1 — CRC8 + Counter (used for safety-critical CAN signals)
struct E2EHeader {
    uint8_t counter : 4;   // 0–15, increments per transmission
    uint8_t crc     : 8;   // E2E CRC over header + data
};

class E2ETransmitter {
public:
    bool protect(uint8_t* frame, uint8_t length) {
        if (frame == nullptr || length < 2U) return false;

        frame[0] = counter_ & 0x0FU;  // Counter in low nibble
        frame[1] = 0x00U;              // CRC placeholder

        // Compute CRC over entire frame (with 0x00 placeholder)
        frame[1] = compute_crc8_j1850(frame, length);

        counter_ = (counter_ + 1U) % 16U;
        return true;
    }

private:
    uint8_t counter_ = 0U;

    static uint8_t compute_crc8_j1850(const uint8_t* data, uint8_t len) {
        uint8_t crc = 0xFFU;
        for (uint8_t i = 0U; i < len; ++i) {
            crc ^= data[i];
            for (uint8_t b = 0U; b < 8U; ++b) {
                crc = (crc & 0x80U) ? ((crc << 1U) ^ 0x1DU) : (crc << 1U);
            }
        }
        return crc ^ 0xFFU;
    }
};

class E2EReceiver {
public:
    enum class Status { OK, WRONG_CRC, WRONG_COUNTER, REPEATED, NO_NEW_DATA };

    Status check(const uint8_t* frame, uint8_t length) {
        // Verify CRC
        uint8_t expected_crc = frame[1];
        uint8_t frame_copy[16U];
        memcpy(frame_copy, frame, length);
        frame_copy[1] = 0x00U;
        uint8_t computed_crc = compute_crc8_j1850(frame_copy, length);

        if (computed_crc != expected_crc) return Status::WRONG_CRC;

        // Check counter
        uint8_t recv_counter = frame[0] & 0x0FU;
        uint8_t expected = (last_counter_ + 1U) % 16U;

        if (recv_counter == last_counter_) return Status::REPEATED;
        if (recv_counter != expected)      return Status::WRONG_COUNTER;

        last_counter_ = recv_counter;
        return Status::OK;
    }

private:
    uint8_t last_counter_ = 0xFFU;  // Init to invalid
    static uint8_t compute_crc8_j1850(const uint8_t* data, uint8_t len);
};
```

## 5. NVM — Non-Volatile Memory Manager

```cpp
// NVM API — persist BMS data across power cycles
#include "NvM.h"

// NvM Block IDs — configured in NvM_Cfg.h
constexpr NvM_BlockIdType NVM_BLOCK_BMS_SOC_INIT   = 0x0001U;
constexpr NvM_BlockIdType NVM_BLOCK_BMS_SOH        = 0x0002U;
constexpr NvM_BlockIdType NVM_BLOCK_BMS_CYCLES     = 0x0003U;
constexpr NvM_BlockIdType NVM_BLOCK_BMS_CALIBRATION = 0x0004U;

struct BmsNvmData {
    float    soc_at_poweroff;     // Last known SoC (for init on next power on)
    float    soh_pct;
    uint32_t total_cycles;
    uint64_t total_energy_wh;
    uint8_t  crc8;                // Integrity check
};

class BmsNvmManager {
public:
    static bool save(const BmsNvmData& data) {
        // Write to RAM mirror (NvM handles actual write on shutdown)
        memcpy(s_nvm_mirror_, &data, sizeof(BmsNvmData));
        Std_ReturnType ret = NvM_WriteBlock(NVM_BLOCK_BMS_SOC_INIT, s_nvm_mirror_);
        return ret == E_OK;
    }

    static bool load(BmsNvmData& data) {
        Std_ReturnType ret = NvM_ReadBlock(NVM_BLOCK_BMS_SOC_INIT, s_nvm_mirror_);
        if (ret != E_OK) return false;
        memcpy(&data, s_nvm_mirror_, sizeof(BmsNvmData));
        return verify_crc(data);
    }

private:
    static uint8_t s_nvm_mirror_[sizeof(BmsNvmData)];

    static bool verify_crc(const BmsNvmData& data) {
        uint8_t computed = compute_crc8(reinterpret_cast<const uint8_t*>(&data),
                                         sizeof(BmsNvmData) - 1U);
        return computed == data.crc8;
    }
};
```

## 7. ISO 26262 — Safety Mechanisms in C++

### Redundant Calculation (ASIL B → ASIL D with redundancy)

```cpp
// Safety goal: OV2 protection must not fail undetected (ASIL B)
// Mechanism: Two independent OV checks with diverse implementation

class PrimaryOvProtection {
public:
    bool is_ov2_detected(uint16_t cell_mv) const {
        return cell_mv > 4250U;     // Direct threshold comparison
    }
};

class DiverseOvProtection {
public:
    bool is_ov2_detected(uint16_t cell_mv) const {
        // Diverse: check via saturation counter with inverted threshold
        static uint8_t counter = 0U;
        if (cell_mv >= 4251U) {
            ++counter;
        } else {
            counter = 0U;
        }
        return counter >= 3U;      // 3-consecutive cycles
    }
};

class SafeOvProtection {
public:
    bool is_ov2_fault(uint16_t cell_mv) {
        bool primary  = primary_.is_ov2_detected(cell_mv);
        bool diverse  = diverse_.is_ov2_detected(cell_mv);

        if (primary != diverse) {
            // Diagnostic: implementations disagree
            DEM_SetEventStatus(DEM_EVENT_OV_PROTECTION_MISMATCH, DEM_EVENT_STATUS_FAILED);
        }

        // OR-of-monitors: flag if either detects fault (fail-safe)
        return primary || diverse;
    }

private:
    PrimaryOvProtection primary_;
    DiverseOvProtection diverse_;
};
```

### Safety Mechanism: Plausibility Check

```cpp
// Cross-check CAN signal against hardware ADC reading (ISO 26262 SM)
bool validate_pack_voltage(float can_voltage_v, uint16_t adc_raw) {
    float adc_voltage_v = static_cast<float>(adc_raw) * (500.0f / 4095.0f);  // 0–500V range

    float deviation = fabsf(can_voltage_v - adc_voltage_v);
    constexpr float PLAUSIBILITY_THRESHOLD_V = 5.0f;

    if (deviation > PLAUSIBILITY_THRESHOLD_V) {
        Dem_SetEventStatus(DEM_EVENT_PACK_VOLTAGE_PLAUSIBILITY, DEM_EVENT_STATUS_FAILED);
        return false;
    }
    return true;
}
```

### Safety Mechanism: Program Flow Monitoring

```cpp
// Execution token — verify all critical functions called in sequence
class ProgramFlowMonitor {
public:
    constexpr static uint32_t TOKEN_READ_SENSORS  = 0x00000001UL;
    constexpr static uint32_t TOKEN_CHECK_VOLTAGE = 0x00000002UL;
    constexpr static uint32_t TOKEN_CHECK_THERMAL = 0x00000004UL;
    constexpr static uint32_t TOKEN_CALC_SOC      = 0x00000008UL;
    constexpr static uint32_t TOKEN_TRANSMIT_CAN  = 0x00000010UL;
    constexpr static uint32_t EXPECTED_TOKEN      = 0x0000001FUL;

    void set_token(uint32_t token) { accumulated_ |= token; }

    bool is_complete(void) const { return accumulated_ == EXPECTED_TOKEN; }

    void reset(void) {
        if (!is_complete()) {
            // Incomplete flow — safety reaction
            Dem_SetEventStatus(DEM_EVENT_PROGRAM_FLOW_ERROR, DEM_EVENT_STATUS_FAILED);
        }
        accumulated_ = 0U;
    }

private:
    uint32_t accumulated_ = 0U;
};

ProgramFlowMonitor flow_monitor;

void BmsMonitor_MainFunction(void) {
    flow_monitor.reset();

    read_sensors();           flow_monitor.set_token(ProgramFlowMonitor::TOKEN_READ_SENSORS);
    check_voltage();          flow_monitor.set_token(ProgramFlowMonitor::TOKEN_CHECK_VOLTAGE);
    check_thermal();          flow_monitor.set_token(ProgramFlowMonitor::TOKEN_CHECK_THERMAL);
    calculate_soc();          flow_monitor.set_token(ProgramFlowMonitor::TOKEN_CALC_SOC);
    transmit_can();           flow_monitor.set_token(ProgramFlowMonitor::TOKEN_TRANSMIT_CAN);
}
```

## 8. ASIL Decomposition in C++

```cpp
// ASIL B decomposed into ASIL A + ASIL A (QM) with independence
// Two channels running on separate code paths (even separate cores on multi-core MCUs)

// Channel A — primary (ASIL A, compiled with different toolchain flags)
namespace ChannelA {
    bool check_ov2(uint16_t mv) { return mv > OV2_THRESHOLD_MV; }
}

// Channel B — diverse monitor (ASIL A, different algorithm)
namespace ChannelB {
    bool check_ov2(uint16_t mv) {
        // Diverse: use floating point comparison
        float voltage = static_cast<float>(mv) / 1000.0f;
        return voltage > 4.250001f;  // Slightly different to catch implementation errors
    }
}

// Safety comparator (QM — compares outputs of both channels)
bool ov2_protected(uint16_t mv) {
    bool a = ChannelA::check_ov2(mv);
    bool b = ChannelB::check_ov2(mv);

    if (a != b) {
        log_message(LogLevel::ERROR, "[SAFETY] Channel mismatch OV2: A=%d B=%d", a, b);
        // Conservative: act on fault if either channel says fault
    }
    return a || b;
}
```
